class villanos:
    

    def __init__(self,nombre,salud,ataque,escudo):
        self.nombre = nombre
        self.salud = salud
        self.ataque = ataque
        self.escudo = escudo

    def __str__(self):
        return self.nombre

    def atacar (self, jugador):
        if jugador.escudo:
            self.salud -= self.ataque



        else:
            jugador.salud -= self.ataque
        print("{} ataca a {}".format(self.nombre, jugador))
        print("{} tiene de salud {}".format(self, self.salud))
        print("{} tiene de salud {}".format(jugador, jugador.salud))
